

-- 1. CRIAÇÃO DO BANCO DE DADOS

CREATE DATABASE IF NOT EXISTS clientesdb;


USE clientesdb;

-- 2. CRIAÇÃO DA TABELA 

CREATE TABLE usuarios (
   
    id BIGINT NOT NULL AUTO_INCREMENT,
    nome VARCHAR(255),
    email VARCHAR(255),
    senha VARCHAR(255),
  
    PRIMARY KEY (id)
);

-- 3. CRIAÇÃO DA TABELA 

CREATE TABLE empreendedores_model (

    id BIGINT NOT NULL AUTO_INCREMENT,
    tipo VARCHAR(255),
    nome_empreendedor VARCHAR(255), 
    item VARCHAR(255),
    descricao VARCHAR(255),
    categoria VARCHAR(255),
    preco VARCHAR(255),
    cidade VARCHAR(255),
    imagem VARCHAR(255),
    

    PRIMARY KEY (id)
);
